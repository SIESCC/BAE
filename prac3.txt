<body ng-controller="formController">

<div class="form-box">

    <h2>Student Form</h2>

    <form name="studentForm" novalidate ng-submit="submitForm(studentForm.$valid)">

        <!-- Name -->
        <input type="text" name="name" ng-model="student.name"
               placeholder="Enter Name"
               required ng-pattern="/^[a-zA-Z ]+$/">

        <span class="error"
              ng-show="studentForm.name.$touched && studentForm.name.$invalid">
            Enter valid name
        </span>

        <!-- Email -->
        <input type="email" name="email" ng-model="student.email"
               placeholder="Enter Email"
               required>

        <span class="error"
              ng-show="studentForm.email.$touched && studentForm.email.$invalid">
            Enter valid email
        </span>

        <!-- Phone -->
        <input type="text" name="phone" ng-model="student.phone"
               placeholder="Enter Phone"
               required ng-pattern="/^[0-9]{10}$/">

        <span class="error"
              ng-show="studentForm.phone.$touched && studentForm.phone.$invalid">
            Enter valid 10-digit number
        </span>

        <!-- Gender -->
        <select name="gender" ng-model="student.gender" required>
            <option value="">Select Gender</option>
            <option>Male</option>
            <option>Female</option>
        </select>

        <span class="error"
              ng-show="studentForm.gender.$touched && studentForm.gender.$invalid">
            Gender is required
        </span>

        <!-- Submit -->
        <button type="submit">Submit</button>

    </form>

</div>

<script>
    var app = angular.module("myApp", []);

    app.controller("formController", function($scope){

        $scope.submitForm = function(isValid){
            if(isValid){
                alert("Form submitted successfully!");
            } else {
                alert("Please fill all required fields correctly!");
            }
        };

    });
</script>

</body>
