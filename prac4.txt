    <form name="myForm" ng-submit="submitForm()" novalidate>

        <!-- Full Name -->
        <input type="text" name="fullname" ng-model="formData.fullname"
               placeholder="Enter Full Name"
               required has-vowel>

        <span class="error"
              ng-show="myForm.fullname.$touched && myForm.fullname.$error.required">
            Name is required
        </span>

        <span class="error"
              ng-show="myForm.fullname.$touched && myForm.fullname.$error.vowel">
            Name must contain at least one vowel
        </span>

        <!-- Email -->
        <input type="email" name="email" ng-model="formData.email"
               placeholder="Enter Email" required>

        <span class="error"
              ng-show="myForm.email.$touched && myForm.email.$invalid">
            Enter valid email
        </span>

        <!-- Password -->
        <input type="password" name="password" ng-model="formData.password"
               placeholder="Enter Password"
               required strong-password>

        <span class="error"
              ng-show="myForm.password.$touched && myForm.password.$error.required">
            Password is required
        </span>

        <span class="error"
              ng-show="myForm.password.$touched && myForm.password.$error.strong">
            Password must contain letter + number
        </span>

<script>
var app = angular.module("myApp", []);

app.directive('hasVowel', function(){
    return {
        require: 'ngModel',
        link: function(scope, element, attrs, ctrl){
            ctrl.$parsers.push(function(value){
                if(/[aeiouAEIOU]/.test(value)){
                    ctrl.$setValidity('vowel', true);
                } else {
                    ctrl.$setValidity('vowel', false);
                }
                return value;
            });
        }
    };
});

app.directive('strongPassword', function(){
    return {
        require: 'ngModel',
        link: function(scope, element, attrs, ctrl){
            ctrl.$parsers.push(function(value){
                if(/^(?=.*[A-Za-z])(?=.*\d).+$/.test(value)){
                    ctrl.$setValidity('strong', true);
                } else {
                    ctrl.$setValidity('strong', false);
                }
                return value;
            });
        }
    };
});

app.controller("myCtrl", function($scope){
    $scope.submitForm = function(){
        if($scope.myForm.$valid){
            alert("Form submitted successfully!");
        } else {
            alert("Please fix validation errors!");
        }
    };
});
</script>
