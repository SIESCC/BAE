<body ng-controller="filterController">

    <div class="container">

        <h1>AngularJS Filters Demonstration</h1>

        <h2>Lowercase Filter</h2>
        <input type="text" ng-model="lowerText" placeholder="Enter text">
        <p>Output: {{ lowerText | lowercase }}</p>

        <h2>Uppercase Filter</h2>
        <input type="text" ng-model="upperText" placeholder="Enter text">
        <p>Output: {{ upperText | uppercase }}</p>

        <h2>Search & OrderBy Filter</h2>
        <input type="text" ng-model="searchText" placeholder="Search fruit">

        <select ng-model="sortColumn">
            <option value="">Sort By</option>
            <option value="name">Name</option>
            <option value="price">Price</option>
        </select>

        <ul>
            <li ng-repeat="fruit in fruits | filter:searchText | orderBy:sortColumn">
                {{ fruit.name }} - {{ fruit.price | currency }}
            </li>
        </ul>

        <h2>Currency Filter</h2>
        <p>Price: {{ productPrice | currency }}</p>

        <h2>Date Filter</h2>
        <p>Full Date: {{ today | date:'fullDate' }}</p>
        <p>Short Date: {{ today | date:'dd/MM/yyyy' }}</p>

        <h2>JSON Filter</h2>
        <pre>{{ product | json }}</pre>

        <h2>LimitTo Filter</h2>
        <input type="number" ng-model="limitValue">
        <p>{{ message | limitTo:limitValue }}</p>

        <h2>Number Filter</h2>
        <p>Original Number: {{ bigNumber }}</p>
        <p>Formatted Number: {{ bigNumber | number:2 }}</p>

    </div>

    <script>
        var app = angular.module("filterApp", []);

        app.controller("filterController", function ($scope) {
            $scope.lowerText = "";
            $scope.upperText = "";
            $scope.productPrice = 2500;
            $scope.today = new Date();
            $scope.bigNumber = 12345.6789;
            $scope.message = "Hello AngularJS Filters Example";
            $scope.limitValue = 10;

            $scope.product = {
                name: "Laptop",
                price: 50000,
                brand: "Dell"
            };

            $scope.fruits = [
                { name: "Apple", price: 1.2 },
                { name: "Banana", price: 0.5 },
                { name: "Cherry", price: 3 },
                { name: "Date", price: 2.5 },
                { name: "Mango", price: 1.8 }
            ];
        });
    </script>
