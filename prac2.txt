        <h3>Add Student</h3>
        <p>Here you can add student details like Name and Roll Number.</p>
    </script>

    <script type="text/ng-template" id="viewStudents.html">
        <h3>View Students</h3>
        <p>Here you can view all registered students.</p>

<script>
    var app = angular.module("mainApp", ['ngRoute']);

    app.config(function($routeProvider){
        $routeProvider
            .when('/addStudent',{
                templateUrl: 'addStudent.html',
                controller: 'AddStudentController'
            })
            .when('/viewStudents',{
                templateUrl: 'viewStudents.html',
                controller: 'ViewStudentsController'
            })
            .otherwise({
                redirectTo: '/addStudent'
            });
    });

    app.controller("AddStudentController", function($scope){
        $scope.message = "Add Student Page";
    });

    app.controller("ViewStudentsController", function($scope){
        $scope.message = "View Students Page";
    });
</script>
